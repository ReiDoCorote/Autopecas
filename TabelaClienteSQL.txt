CREATE DATABASE autopecas;
CREATE TABLE Juridica (
  idJuridica INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  CNPJ VARCHAR(20) NULL,
  Nome VARCHAR(75) NULL,
  Endereco VARCHAR(75) NULL,
  Cidade VARCHAR(75) NULL,
  Bairro VARCHAR(75) NULL,
  CEP VARCHAR(9) NULL,
  Telefone VARCHAR(14) NULL,
  Celular VARCHAR(15) NULL,
  Email VARCHAR(75) NULL,
  LimiteDeCredito FLOAT NULL,
  PRIMARY KEY(idJuridica)
);

CREATE TABLE Fisica (
  idFisica INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  CPF VARCHAR(20) NULL,
  RG VARCHAR(20) NULL,
  Nome VARCHAR(75) NULL,
  Endereco VARCHAR(75) NULL,
  Cidade VARCHAR(75) NULL,
  Bairro VARCHAR(75) NULL,
  CEP VARCHAR(9) NULL,
  Telefone VARCHAR(14) NULL,
  Celular VARCHAR(15) NULL,
  Email VARCHAR(75) NULL,
  LimiteDeCredito FLOAT NULL,
  PRIMARY KEY(idFisica)
);
