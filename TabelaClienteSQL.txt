CREATE TABLE Cliente (
  CPF/CNPJ INTEGER(15) UNSIGNED NOT NULL,
  Nome VARCHAR NULL,
  Endereco VARCHAR NULL,
  Cidade VARCHAR NULL,
  Bairro VARCHAR NULL,
  CEP VARCHAR NULL,
  Telefone VARCHAR NULL,
  Celular VARCHAR NULL,
  Email VARCHAR NULL,
  LimiteDeCredito FLOAT NULL,
  RG VARCHAR NULL,
  PRIMARY KEY(CPF/CNPJ)
);
